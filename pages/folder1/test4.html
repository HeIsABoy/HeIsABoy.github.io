 <--!(2)鼠标点击动作 在js层要定义clickNum clickNum1并写成箭头的形式 对应(2)-->

  </div>
  
  <--!(3)输出ref的单个数据-->
  <div>
    {{ name }}
  </div>
  <div>
    {{ arr.slice(0, 2) }}
    {{obj}}
  </div>
  
  <--!(4)输入框 键盘enter 确认要加@后的东西 js层(3)-->
   <div>
    <input placeholder="请输入任务名称" v-model="value" @keydown.enter="enter" />
  </div>
  
  <--!(5)前面的选中框-->
   <input type="checkbox" v-model="item.complete" />
   
  <--!(6)循环显示列表 通过index循环索引到item-->
   <div v-for="(item, index) in list" :key="index">
    <div>
      <input type="checkbox" v-model="item.complete" />
      {{ item.title }}
      <button>删除</button>
    </div>
  </div>
  
  <--!(7)判断-->
  <div v-if="isComplete > 0">
      <button>清除已完成</button>
    </div>
</template>
