# 这是一个标记语言

# 1 html 嵌套

<p>
  <strong>
  这是一个段落
  </strong>
</p>

# 2 空元素

<img class="imges/a.png" alt="测试图片">

# 3 文档详解

<!DOCTYPE html>
<html>
  <head>
    <meta charset="utf-8">
    <title>
    测试页面
    </title>
  </head>
  <body>
    <img src="images/firefox-icon.png" alt="测试图片">
  </body>
</html>
