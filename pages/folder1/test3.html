# JS

# 动态编程语言

# 1 变量 let或var 声明变量

let myVariable = '李雷';
myVariable = '韩梅梅';

变量数据类型有5种

  1. String
  2. Number
  3. Boolean
  4. Array
  5. Object:let myVariable = document.querySelector('h1');

# 2 if else与java相同

let iceCream = 'chocolate';
if (iceCream === 'chocolate') {
  alert('我最喜欢巧克力冰激淋了。');
} else {
  alert('但是巧克力才是我的最爱呀……');
}

# 3 方法定义也与java差不多

function multiply(num1, num2) {
  let result = num1 * num2;
  return result;
}

# 4点击事件

document.querySelector('html').onclick = function() {
    alert('别戳我，我怕疼。');
}
